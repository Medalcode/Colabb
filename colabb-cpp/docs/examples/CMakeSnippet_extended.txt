-- CMake snippet para compilar un plugin multiplataforma

add_library(example_provider SHARED skill_skeleton_extended.cpp)
target_include_directories(example_provider PRIVATE ${CMAKE_SOURCE_DIR}/include)

if (WIN32)
  set_target_properties(example_provider PROPERTIES SUFFIX ".dll")
elseif(APPLE)
  set_target_properties(example_provider PROPERTIES SUFFIX ".dylib")
else()
  set_target_properties(example_provider PROPERTIES SUFFIX ".so")
endif()
