# CMake Snippet — Cómo compilar un Plugin para Colabb (Multiplataforma)

add_library(skeleton_provider SHARED skill_skeleton.cpp)

# Asegurar que los headers del dominio sean visibles
target_include_directories(skeleton_provider PRIVATE 
    ${CMAKE_SOURCE_DIR}/src
    ${CMAKE_SOURCE_DIR}/include
)

# Ajuste automático del sufijo según el sistema operativo
if (WIN32)
    set_target_properties(skeleton_provider PROPERTIES SUFFIX ".dll")
elseif(APPLE)
    set_target_properties(skeleton_provider PROPERTIES SUFFIX ".dylib")
else()
    set_target_properties(skeleton_provider PROPERTIES SUFFIX ".so")
endif()

# Mensaje de ayuda post-compilación
message(STATUS "Plugin simple configurado como: skeleton_provider")
